% Command for plotting a figure of a group of ROC curves:
\newcommand{\MakeAfigure}[1]{
\begin{center}
\begin{tikzpicture}
\begin{groupplot}[
  group style={group size=3 by 4,
  horizontal sep=0pt, vertical sep=0pt},
  width=2.8in,
  height=2.8in,
  domain=0:1,
  restrict y to domain=0:1,
  samples=100,
  minor tick num=1,
  xmin=0, xmax=1,
  ymin=0, ymax=1,
  xticklabels={,,},
  yticklabels={,,},
  tick label style={font={\fontsize{10pt}{12pt}\selectfont}},
  tick style={line width=.25pt, color=black, line cap=round},
  label style={color=axescolor, font={\fontsize{12pt}{14}\selectfont}},
  tick label style={color=axescolor, font={\fontsize{10pt}{12pt}\selectfont}},
  tick style={line width=1.pt, color=axescolor, line cap=round},
  axis line style={color=axescolor, line width=2pt, },
  legend style={
    at={(0.25,0.2)},anchor=west,
    fill=none,
   % draw opacity=0,
    font={\fontsize{10pt}{12pt}\selectfont}},
  set layers=standard,
] #1
\end{groupplot}
\end{tikzpicture}
\end{center}
}
